97 double computeMode(vector<double> &newAngles)
98 {
99  double mode = newAngles[0];
100  int freq = 1;
101  int tempFreq;
102  double diff;
103  for (int i = 0; i < newAngles.size(); i++)
104  {
105  tempFreq = 1;
106 
107  for (int j = i + 1; j < newAngles.size(); j++)
108  {
109  diff = newAngles[j] - newAngles[i] > 0.0 ? newAngles[j] - newAngles[i] : newAngles[i] - newAngles[j];
110  if (diff <= minDeviation)
111  {
112  tempFreq++;
113  newAngles.erase(newAngles.begin() + j);
114  j = j - 1;
115  }
116  }
117 
118  if (tempFreq >= freq)
119  {
120  mode = newAngles[i];
121  freq = tempFreq;
122  }
123  }
124 
125  return mode;
126 }
