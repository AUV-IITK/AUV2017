88 double computeMode(vector<double> &newAngles)
89 {
90  double mode = newAngles[0];
91  int freq = 1;
92  int tempFreq;
93  double diff;
94  for (int i = 0; i < newAngles.size(); i++)
95  {
96  tempFreq = 1;
97 
98  for (int j = i + 1; j < newAngles.size(); j++)
99  {
100  diff = newAngles[j] - newAngles[i] > 0.0 ? newAngles[j] - newAngles[i] : newAngles[i] - newAngles[j];
101  if (diff <= minDeviation)
102  {
103  tempFreq++;
104  newAngles.erase(newAngles.begin() + j);
105  j = j - 1;
106  }
107  }
108 
109  if (tempFreq >= freq)
110  {
111  mode = newAngles[i];
112  freq = tempFreq;
113  }
114  }
115 
116  return mode;
117 }
